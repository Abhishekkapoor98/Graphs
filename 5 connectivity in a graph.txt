bool visited[100005] = {false};
vi graph[100005];
	
void add_edge(int u,int v)
{
	debug(u);
	cerr<<" ";
	debug(v);

	graph[u].PB(v);
	graph[v].PB(u);
}

void dfs(int root)
{	
	debug(root);
	cerr<<"\n";

	if(visited[root])
		return;

	visited[root] = true;
	
	for(auto itr: graph[root])
		dfs(itr);
}	

void solve()
{
	int t;
	cin>>t;
	debug(t);

	while(t--)
	{
		int n,e;
		cin>>n>>e;
		debug(n);
		debug(e);

		while(e--)
		{
			int u,v;
			cin>>u>>v;
			add_edge(u,v);
		}

		dfs(1);

		bool flag = true;
		for(int i=1;i<=n;i++)
		{
			if(!visited[i]){
				cout<<"non-connected graph."<<endl;
				flag = false;
			}
		}

		if(flag)
			cout<<"connected graph."<<endl;
	}
}
